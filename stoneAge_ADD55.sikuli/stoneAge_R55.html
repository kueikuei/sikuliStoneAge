def MemberbattleSetting():
    # 更換寵物狀態
    click(Location(290, 617))

    click(Location(42, 70)) #防呆機制
    wait(0.1)
    while not exists():
        click(Location(42, 70))

    wait(0.1)

    #click(Location(41, 122)) #防呆機制
    wait(0.1)
    while not exists():
        click(Location(41, 122))

    click(systemLocation)

    # 關閉AI
    if exists():
        click()

    wait(0.1)

    # 關閉systemLocation
    click()
